import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

# 1. Load and clean data
df = pd.read_csv("C:\\gfg python\\Research data.csv")
df_clean = df.iloc[:,2:].copy()
df_clean.columns = ["Consent", "Programme", "Level", "Q1_Familiar_AI", "Q2_Follows_AI_News",
                    "Q3_AI_Opportunities", "Q4_Uncertainty_Rev", "Q5_Tech_Skills", "Q6_Soft_Skills",
                    "Q7_Uni_Preparation", "Q8_Confidence_Adapt", "Q9_Field_Relevance", "Q10_Optimism"]
df_clean = df_clean[df_clean['Consent'] == "YES"]

# 2. Explicit category order for x-axis
order = [
    "STEM",
    "Business / Economics",
    "Arts & Humanities",
    "Social Sciences",
    "Other (please specify)"
]

# 3. Calculate means for ordered disciplines
optimism_scores = df_clean.groupby("Programme")["Q10_Optimism"].mean().reindex(order)
anxiety_scores = df_clean.groupby("Programme")["Q4_Uncertainty_Rev"].mean().reindex(order)
soft_skill_scores = df_clean.groupby("Programme")["Q6_Soft_Skills"].mean().reindex(order)
tech_skill_scores = df_clean.groupby("Programme")["Q5_Tech_Skills"].mean().reindex(order)
awareness_mean = df_clean.groupby("Programme")[["Q1_Familiar_AI", "Q2_Follows_AI_News"]].mean().mean(axis=1).reindex(order)
optimism_mean_scatter = optimism_scores

x = np.arange(len(order))
width = 0.35

# Calculate AI awareness and cast columns
for col in ["Q1_Familiar_AI", "Q2_Follows_AI_News", "Q10_Optimism"]:
    df_clean[col] = pd.to_numeric(df_clean[col], errors='coerce')
df_clean['AI_Awareness'] = df_clean[["Q1_Familiar_AI", "Q2_Follows_AI_News"]].mean(axis=1)

# Scatter plot with regression line for ALL students
plt.figure(figsize=(8,5))
sns.regplot(
    x="AI_Awareness",
    y="Q10_Optimism",
    data=df_clean,
    scatter_kws={"s":60, "color":"dodgerblue", "alpha":0.7},
    line_kws={"color":"crimson", "lw":2})
plt.xlabel('AI Awarenes')
plt.ylabel('Optimism (Q12)')
plt.title('Correlation between AI Awareness and Optimism')
plt.grid(True, linestyle='--', alpha=0.4)
plt.tight_layout()

# Show Pearson correlation on the chart
from scipy.stats import pearsonr
corr, _ = pearsonr(df_clean['AI_Awareness'], df_clean['Q10_Optimism'])
plt.text(3.02, df_clean['Q10_Optimism'].min()+0.5, f'Pearson r = {corr:.2f}', fontsize=12, color='crimson')

plt.show()
